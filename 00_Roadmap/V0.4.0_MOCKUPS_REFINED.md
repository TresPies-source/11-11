## Design System (Refined)

### Color Palette (Dojo Genesis)

**Primary Brand Colors:**
```css
/* Backgrounds (Dark Mode) */
--bg-primary:   #0a1e2e;  /* Navy 900 - Main background */
```

### React Components

**AgentCard.tsx:**
```typescript
interface AgentCardProps {
  agent: 'supervisor' | 'dojo' | 'librarian' | 'debugger';
  status: 'idle' | 'thinking' | 'working' | 'complete' | 'error';
  task?: string;
  progress?: number;
  cost?: number;
  duration?: number;
  expanded?: boolean;
}

export function AgentCard({ agent, status, task, progress, cost, duration, expanded }: AgentCardProps) {
  const agentConfig = {
    supervisor: { icon: 'ğŸ¯', name: 'Supervisor', color: 'supervisor' },
    dojo: { icon: 'ğŸ§ ', name: 'Dojo', color: 'dojo' },
    librarian: { icon: 'ğŸ“š', name: 'Librarian', color: 'librarian' },
    debugger: { icon: 'ğŸ”', name: 'Debugger', color: 'debugger' },
  }[agent];

  return (
    <motion.div
      className={`bg-bg-secondary border border-bg-tertiary rounded-xl p-6 ${expanded ? 'w-80' : 'w-20'}`}
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.3 }}
    >
      {/* Agent icon and status */}
      <div className="flex items-center justify-between">
        <span className="text-4xl">{agentConfig.icon}</span>
        <StatusDot status={status} color={agentConfig.color} />
      </div>
      
      {expanded && (
        <>
          <h3 className="text-lg font-medium text-text-primary mt-4">{agentConfig.name}</h3>
          <p className="text-sm text-text-secondary mt-1">Status: {status}</p>
          {task && <p className="text-sm text-text-tertiary mt-2">{task}</p>}
          {progress !== undefined && (
            <ProgressBar progress={progress} color={agentConfig.color} className="mt-3" />
          )}
          {(cost !== undefined || duration !== undefined) && (
            <div className="flex justify-between text-xs text-text-tertiary mt-3">
              {duration !== undefined && <span>Duration: {duration}s</span>}
              {cost !== undefined && <span>Cost: ${cost.toFixed(4)}</span>}
            </div>
          )}
        </>
      )}
    </motion.div>
  );
}
```
